from abc import ABC, abstractmethod

class IInstagramApiPort(ABC):

        pass

        """Fetch Instagram stories."""
        pass
    
    @abstractmethod
    async def fetch_posts(self, username: InstagramUsername, cursor: str | None, limit: int) -> List[Post]:
        """Fetch Instagram posts."""
        pass
